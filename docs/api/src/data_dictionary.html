  • Initial prompt generation for AI models
  • Structured data reference for DShield SIEM

538    @staticmethod
539    def _format_field_section(fields: dict[str, Any]) -> str:
540        """Format field descriptions into a readable section.
541
542        Args:
543            fields: Dictionary containing field descriptions
544
545        Returns:
546            Formatted string representation of the field section
547
548        """
549        formatted = []
550        for field_name, field_info in fields.items():
551            formatted.append(f"- **{field_name}**: {field_info['description']}")
552            if "examples" in field_info:
553                formatted.append(f"  - Examples: {', '.join(map(str, field_info['examples']))}")
554            if "usage" in field_info:
555                formatted.append(f"  - Usage: {field_info['usage']}")
556            formatted.append("")
557        return "\n".join(formatted)
