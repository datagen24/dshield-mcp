src.dshield_client

DShield client for threat intelligence and IP reputation lookup.

This module provides a client for interacting with the DShield threat intelligence API. It supports IP reputation lookups, attack summaries, batch enrichment, and detailed IP information retrieval. The client handles authentication, rate limiting, caching, and error handling for robust integration with DShield services.

Features:

  • IP reputation and details lookup
  • Attack summary retrieval
  • Batch enrichment of IPs
  • Caching and rate limiting
  • Async context management

Example:

from src.dshield_client import DShieldClient async with DShieldClient() as client: ... rep = await client.get_ip_reputation("8.8.8.8") ... print(rep)

170    async def get_ip_reputation(self, ip_address: str) -> Dict[str, Any]:
171        """Get IP reputation from DShield.
172        
173        Looks up the reputation of a given IP address using the DShield API.
174        Utilizes caching and rate limiting for efficiency.
175        
176        Args:
177            ip_address: The IP address to look up
178        
179        Returns:
180            Dictionary containing reputation data for the IP
181        """
182        
183        # Check cache first
184        cache_key = f"reputation_{ip_address}"
185        cached_data = self._get_cached_data(cache_key)
186        if cached_data:
187            logger.debug("Returning cached IP reputation", ip_address=ip_address)
188            return cached_data
189        
190        # Rate limiting
191        await self._check_rate_limit()
192        
193        try:
194            await self.connect()
195            
196            # DShield IP reputation endpoint
197            url = urljoin(self.base_url, f"ip/{ip_address}")
198            
199            logger.info("Querying DShield IP reputation", ip_address=ip_address)
200            
201            async with self.session.get(url) as response:
202                if response.status == 200:
203                    data = await response.json()
204                    reputation_data = self._parse_ip_reputation(data, ip_address)
205                    
206                    # Cache the result
207                    self._cache_data(cache_key, reputation_data)
208                    
209                    logger.info("IP reputation retrieved successfully", 
210                               ip_address=ip_address,
211                               reputation_score=reputation_data.get('reputation_score'))
212                    
213                    return reputation_data
214                    
215                elif response.status == 404:
216                    # IP not found in DShield database
217                    reputation_data = self._create_default_reputation(ip_address)
218                    self._cache_data(cache_key, reputation_data)
219                    return reputation_data
220                    
221                else:
222                    logger.warning("DShield API returned non-200 status", 
223                                  ip_address=ip_address,
224                                  status=response.status)
225                    return self._create_default_reputation(ip_address)
226                    
227        except aiohttp.ClientError as e:
228            logger.error("HTTP error during IP reputation lookup", 
229                        ip_address=ip_address, error=str(e))
230            return self._create_default_reputation(ip_address)
231            
232        except Exception as e:
233            logger.error("Unexpected error during IP reputation lookup", 
234                        ip_address=ip_address, error=str(e))
235            return self._create_default_reputation(ip_address)

344    async def enrich_ips_batch(self, ip_addresses: List[str]) -> Dict[str, Dict[str, Any]]:
345        """Enrich multiple IP addresses with threat intelligence."""
346        
347        results = {}
348        
349        batch_size = self.batch_size
350        for i in range(0, len(ip_addresses), batch_size):
351            batch = ip_addresses[i:i + batch_size]
352            
353            # Process batch concurrently
354            tasks = [self.get_ip_reputation(ip) for ip in batch]
355            batch_results = await asyncio.gather(*tasks, return_exceptions=True)
356            
357            # Store results
358            for ip, result in zip(batch, batch_results):
359                if isinstance(result, Exception):
360                    logger.warning("Failed to enrich IP", ip=ip, error=str(result))
361                    results[ip] = self._create_default_reputation(ip)
362                else:
363                    results[ip] = result
364            
365            # Small delay between batches
366            if i + batch_size < len(ip_addresses):
367                await asyncio.sleep(1)
368        
369        return results

371    def _parse_ip_reputation(self, data: Dict[str, Any], ip_address: str) -> Dict[str, Any]:
372        """Parse DShield IP reputation response."""
373        
374        reputation_data = {
375            'ip_address': ip_address,
376            'reputation_score': None,
377            'threat_level': 'unknown',
378            'country': None,
379            'asn': None,
380            'organization': None,
381            'first_seen': None,
382            'last_seen': None,
383            'attack_types': [],
384            'tags': [],
385            'raw_data': data
386        }
387        
388        try:
389            # Extract reputation score
390            if 'reputation' in data:
391                reputation_data['reputation_score'] = float(data['reputation'])
392                
393                # Determine threat level based on reputation score
394                if reputation_data['reputation_score'] >= 80:
395                    reputation_data['threat_level'] = 'high'
396                elif reputation_data['reputation_score'] >= 50:
397                    reputation_data['threat_level'] = 'medium'
398                else:
399                    reputation_data['threat_level'] = 'low'
400            
401            # Extract geographic information
402            if 'country' in data:
403                reputation_data['country'] = data['country']
404            
405            # Extract ASN information
406            if 'as' in data:
407                reputation_data['asn'] = data['as']
408            
409            # Extract organization
410            if 'org' in data:
411                reputation_data['organization'] = data['org']
412            
413            # Extract timestamps
414            if 'firstseen' in data:
415                reputation_data['first_seen'] = data['firstseen']
416            
417            if 'lastseen' in data:
418                reputation_data['last_seen'] = data['lastseen']
419            
420            # Extract attack types
421            if 'attacks' in data:
422                reputation_data['attack_types'] = data['attacks']
423            
424            # Extract tags
425            if 'tags' in data:
426                reputation_data['tags'] = data['tags']
427            
428        except Exception as e:
429            logger.warning("Failed to parse IP reputation data", 
430                          ip_address=ip_address, error=str(e))
431        
432        return reputation_data

538    def _get_cached_data(self, cache_key: str) -> Optional[Dict[str, Any]]:
539        """Get data from cache if not expired."""
540        if cache_key in self.cache:
541            cached_item = self.cache[cache_key]
542            if time.time() - cached_item['timestamp'] < self.cache_ttl:
543                return cached_item['data']
544            else:
545                # Remove expired cache entry
546                del self.cache[cache_key]
547        return None

550        """Cache data with timestamp."""

556    async def _check_rate_limit(self):
557        """Check and enforce rate limiting."""
558        current_time = time.time()
559        
560        # Remove old request times outside the window
561        self.request_times = [t for t in self.request_times 
562                            if current_time - t < self.rate_limit_window]
563        
564        # Check if we're at the rate limit
565        if len(self.request_times) >= self.rate_limit_requests:
566            # Calculate wait time
567            oldest_request = min(self.request_times)
568            wait_time = self.rate_limit_window - (current_time - oldest_request)
569            
570            if wait_time > 0:
571                logger.info("Rate limit reached, waiting", wait_time=wait_time)
572                await asyncio.sleep(wait_time)
573        
574        # Add current request time
575        self.request_times.append(current_time)

345    async def enrich_ips_batch(self, ip_addresses: List[str]) -> Dict[str, Dict[str, Any]]:
346        """Enrich multiple IP addresses with threat intelligence."""
347        
348        results = {}
349        
350        batch_size = self.batch_size
351        for i in range(0, len(ip_addresses), batch_size):
352            batch = ip_addresses[i:i + batch_size]
353            
354            # Process batch concurrently
355            tasks = [self.get_ip_reputation(ip) for ip in batch]
356            batch_results = await asyncio.gather(*tasks, return_exceptions=True)
357            
358            # Store results
359            for ip, result in zip(batch, batch_results):
360                if isinstance(result, Exception):
361                    logger.warning("Failed to enrich IP", ip=ip, error=str(result))
362                    results[ip] = self._create_default_reputation(ip)
363                else:
364                    results[ip] = result
365            
366            # Small delay between batches
367            if i + batch_size < len(ip_addresses):
368                await asyncio.sleep(1)
369        
370        return results

372    def _parse_ip_reputation(self, data: Dict[str, Any], ip_address: str) -> Dict[str, Any]:
373        """Parse DShield IP reputation response."""
374        
375        reputation_data = {
376            'ip_address': ip_address,
377            'reputation_score': None,
378            'threat_level': 'unknown',
379            'country': None,
380            'asn': None,
381            'organization': None,
382            'first_seen': None,
383            'last_seen': None,
384            'attack_types': [],
385            'tags': [],
386            'raw_data': data
387        }
388        
389        try:
390            # Extract reputation score
391            if 'reputation' in data:
392                reputation_data['reputation_score'] = float(data['reputation'])
393                
394                # Determine threat level based on reputation score
395                if reputation_data['reputation_score'] >= 80:
396                    reputation_data['threat_level'] = 'high'
397                elif reputation_data['reputation_score'] >= 50:
398                    reputation_data['threat_level'] = 'medium'
399                else:
400                    reputation_data['threat_level'] = 'low'
401            
402            # Extract geographic information
403            if 'country' in data:
404                reputation_data['country'] = data['country']
405            
406            # Extract ASN information
407            if 'as' in data:
408                reputation_data['asn'] = data['as']
409            
410            # Extract organization
411            if 'org' in data:
412                reputation_data['organization'] = data['org']
413            
414            # Extract timestamps
415            if 'firstseen' in data:
416                reputation_data['first_seen'] = data['firstseen']
417            
418            if 'lastseen' in data:
419                reputation_data['last_seen'] = data['lastseen']
420            
421            # Extract attack types
422            if 'attacks' in data:
423                reputation_data['attack_types'] = data['attacks']
424            
425            # Extract tags
426            if 'tags' in data:
427                reputation_data['tags'] = data['tags']
428            
429        except Exception as e:

431                          ip_address=ip_address, error=str(e))

557    async def _check_rate_limit(self):
558        """Check and enforce rate limiting."""
559        current_time = time.time()
560        
561        # Remove old request times outside the window
562        self.request_times = [t for t in self.request_times 
563                            if current_time - t < self.rate_limit_window]
564        
565        # Check if we're at the rate limit
566        if len(self.request_times) >= self.rate_limit_requests:
567            # Calculate wait time
568            oldest_request = min(self.request_times)
569            wait_time = self.rate_limit_window - (current_time - oldest_request)
570            
571            if wait_time > 0:
572                logger.info("Rate limit reached, waiting", wait_time=wait_time)
573                await asyncio.sleep(wait_time)
574        
575        # Add current request time
576        self.request_times.append(current_time)
