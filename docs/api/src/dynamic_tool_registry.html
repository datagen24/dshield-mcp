106    def __init__(self, feature_manager: FeatureManager) -> None:
107        """Initialize the dynamic tool registry.
108
109        Args:
110            feature_manager: Feature manager instance
111
112        """
113        self.feature_manager = feature_manager
114        self.available_tools: list[str] = []
115        self.tool_details: dict[str, dict[str, Any]] = {}
116        self._initialized = False
117
118    def register_tools(self, all_tools: list[str]) -> list[str]:
119        """Register tools based on available features.
120
121        Args:
122            all_tools: List of all available tool names
123
124        Returns:
125            List[str]: List of available tool names
126
127        """
128        try:
129            logger.info("Registering tools based on feature availability")
130
131            self.available_tools = []
132            self.tool_details = {}
133
134            for tool_name in all_tools:
135                # Get the feature this tool depends on
136                feature = TOOL_FEATURE_MAPPING.get(tool_name, "unknown")
137
138                # Check if the feature is available
139                if feature == "unknown":
140                    # Tool not mapped - assume available
141                    is_available = True
142                    reason = "No dependency mapping found - assuming available"
143                    feature_status = "unknown"
144                else:
145                    is_available = self.feature_manager.is_feature_available(feature)
146                    if is_available:
147                        reason = f"Feature '{feature}' is available"
148                        feature_status = "healthy"
149                    else:
150                        reason = f"Feature '{feature}' is unavailable"
151                        feature_status = "unhealthy"
152
153                # Store tool details
154                self.tool_details[tool_name] = {
155                    "available": is_available,
156                    "feature": feature,
157                    "feature_status": feature_status,
158                    "reason": reason,
159                    "description": TOOL_DESCRIPTIONS.get(tool_name, "No description available"),
160                    "dependencies": self.feature_manager.get_feature_dependencies(feature)
161                    if feature != "unknown"
162                    else [],
163                }
164
165                if is_available:
166                    self.available_tools.append(tool_name)
167
168            self._initialized = True
169
170            # Log tool registration summary
171            available_count = len(self.available_tools)
172            total_count = len(all_tools)
173
174            logger.info(
175                "Tool registration completed",
176                available_tools=available_count,
177                total_tools=total_count,
178                availability_percentage=round(available_count / total_count * 100, 1),
179            )
180
181            return self.available_tools
182
183        except Exception as e:
184            logger.error("Tool registration failed", error=str(e))
185            # Return empty list if registration fails
186            return []

188    def get_tool_availability(self) -> dict[str, bool]:
189        """Get availability status for all tools.
190
191        Returns:
192            Dict[str, bool]: Dictionary mapping tool names to availability status
193
194        """
195        if not self._initialized:
196            return {}
197
198        return {tool: (tool in self.available_tools) for tool in self.tool_details.keys()}

250    def get_tool_summary(self) -> dict[str, Any]:
251        """Get a summary of tool availability.
252
253        Returns:
254            Dict[str, Any]: Tool availability summary

256        """
257        if not self._initialized:
258            return {

267        total_count = len(self.tool_details)

307    def refresh_tool_status(self) -> None:
308        """Refresh tool status based on current feature availability.
309
310        This method can be called to update tool availability
311        without re-registering all tools.
312        """
313        if self._initialized and self.tool_details:
314            # Re-evaluate tool availability based on current feature status
315            for tool_name, tool_detail in self.tool_details.items():
316                feature = tool_detail.get("feature")
317                if feature and feature != "unknown":
318                    is_available = self.feature_manager.is_feature_available(feature)
319                    tool_detail["available"] = is_available
320                    tool_detail["feature_status"] = "healthy" if is_available else "unhealthy"
321                    tool_detail["reason"] = (
322                        f"Feature '{feature}' is {'available' if is_available else 'unavailable'}"
323                    )
324
325                    if is_available and tool_name not in self.available_tools:
326                        self.available_tools.append(tool_name)
327                    elif not is_available and tool_name in self.available_tools:
328                        self.available_tools.remove(tool_name)

122        Args:
123            all_tools: List of all available tool names
124
125        Returns:
126            List[str]: List of available tool names
127
128        """
129        try:
130            logger.info("Registering tools based on feature availability")
131
132            self.available_tools = []
133            self.tool_details = {}
134
135            for tool_name in all_tools:
136                # Get the feature this tool depends on
137                feature = TOOL_FEATURE_MAPPING.get(tool_name, "unknown")
138
139                # Check if the feature is available
140                if feature == "unknown":
141                    # Tool not mapped - assume available
142                    is_available = True
143                    reason = "No dependency mapping found - assuming available"
144                    feature_status = "unknown"
145                else:
146                    is_available = self.feature_manager.is_feature_available(feature)
147                    if is_available:
148                        reason = f"Feature '{feature}' is available"
149                        feature_status = "healthy"
150                    else:
151                        reason = f"Feature '{feature}' is unavailable"
152                        feature_status = "unhealthy"
153
154                # Store tool details
155                self.tool_details[tool_name] = {
156                    "available": is_available,
157                    "feature": feature,
158                    "feature_status": feature_status,
159                    "reason": reason,
160                    "description": TOOL_DESCRIPTIONS.get(tool_name, "No description available"),
161                    "dependencies": self.feature_manager.get_feature_dependencies(feature)
162                    if feature != "unknown"
163                    else [],
164                }
165
166                if is_available:
167                    self.available_tools.append(tool_name)
168
169            self._initialized = True
170
171            # Log tool registration summary
172            available_count = len(self.available_tools)
173            total_count = len(all_tools)
174
175            logger.info(
176                "Tool registration completed",
177                available_tools=available_count,
178                total_tools=total_count,
179                availability_percentage=round(available_count / total_count * 100, 1),
180            )
181
182            return self.available_tools
183
184        except Exception as e:
185            logger.error("Tool registration failed", error=str(e))
186            # Return empty list if registration fails
