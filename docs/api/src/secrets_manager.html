293    def _is_cache_valid(self, cache_entry: tuple[Any, datetime]) -> bool:
294        """Check if a cache entry is still valid based on TTL.
295
296        Args:
297            cache_entry: Tuple of (value, timestamp)
298
299        Returns:
300            True if the cache entry is still valid, False otherwise
301        """
302        if not self._enable_caching:
303            return False
304        value, timestamp = cache_entry
305        age = (datetime.now(UTC) - timestamp).total_seconds()
306        return age < self._cache_ttl_seconds
307
308    def _get_from_cache(self, key: str) -> Any | None:
309        """Retrieve a value from the cache if valid.
310
311        Args:
312            key: The cache key to retrieve
313
314        Returns:
315            Cached value if valid, None otherwise
316        """
317        if not self._enable_caching:
318            return None
319
320        normalized_key = self._normalize_cache_key(key)
321        if normalized_key in self._cache:
322            cache_entry = self._cache[normalized_key]
323            if self._is_cache_valid(cache_entry):
324                return cache_entry[0]
325            else:
326                # Remove expired entry
327                del self._cache[normalized_key]
328        return None

331        """Store a value in the cache.

336        """
337        if not self._enable_caching:

 95    def _discover_session(self) -> None:
 96        """Discover existing 1Password session from environment variables.
 97
 98        Looks for OP_SESSION_* environment variables and extracts session tokens.
 99        """
100        session_vars = {k: v for k, v in os.environ.items() if k.startswith("OP_SESSION_")}
101
102        if session_vars:
103            # Use the first available session
104            session_var, session_token = next(iter(session_vars.items()))
105            self._session_token = session_token
106            self.logger.debug(f"Discovered session from {session_var}")
107
108            # Try to get session expiry info
109            try:
110                result = self._run_op_command_sync(["whoami"], timeout=5)
111                if result and "expires_at" in result:
112                    self._session_expires_at = datetime.fromisoformat(
113                        result["expires_at"].replace("Z", "+00:00")
114                    )
115            except Exception:
116                # If we can't get expiry info, assume session is valid for now
117                self._session_expires_at = datetime.now(UTC).replace(hour=23, minute=59, second=59)
118        else:
119            self.logger.debug("No OP_SESSION_* environment variables found")

164    def _run_op_command_sync(self, args: list[str], timeout: int | None = None) -> Any:
165        """Run op CLI command synchronously and return JSON output.
166
167        Args:
168            args: List of arguments to pass to op CLI
169            timeout: Override default timeout in seconds
170
171        Returns:
172            Parsed JSON output from the command
173
174        Raises:
175            SecretsManagerError: For various error conditions
176
177        """
178        timeout_seconds = timeout or self.timeout_seconds
179        cmd = ["op", *args]
180        
181        # Only add --format json for commands that support it
182        if args and args[0] not in ["--version", "completion"]:
183            cmd.extend(["--format", "json"])
184
185        # Add session token if available
186        if self._session_token:
187            cmd.extend(["--session", self._session_token])
188
189        start_time = time.time()
190        latency_ms = 0.0
191        result = None
192
193        try:
194            # Log command with redacted sensitive information
195            safe_cmd = self._redact_sensitive_args(cmd)
196            self.logger.debug(f"Running op command: {' '.join(safe_cmd)}")
197
198            result = subprocess.run(
199                cmd,
200                check=False,
201                capture_output=True,
202                text=True,
203                timeout=timeout_seconds,
204                shell=False,  # Security: prevent shell injection
205            )
206
207            latency_ms = (time.time() - start_time) * 1000
208
209            if result.returncode != 0:
210                error_msg = result.stderr.strip() or "Unknown op CLI error"
211                self.logger.error(f"op command failed (exit {result.returncode}): {error_msg}")
212
213                # Map specific error conditions to appropriate exceptions
214                mapped_error = self._map_op_error(result.returncode, error_msg)
215                if mapped_error:
216                    raise mapped_error
217                else:
218                    raise SecretsManagerError(f"op command failed: {error_msg}")
219
220            if not result.stdout.strip():
221                return {}
222
223            # For non-JSON commands, return the raw output
224            if args and args[0] in ["--version", "completion"]:
225                return result.stdout.strip()
226
227            # Parse and validate JSON output
228            try:
229                json_output = json.loads(result.stdout)
230                self._validate_op_output(json_output, args)
231                return json_output
232            except json.JSONDecodeError as e:
233                self.logger.error(f"Failed to parse op CLI JSON output: {e}")
234                self.logger.debug(f"Raw output: {result.stdout}")
235                raise SecretsManagerError(f"Failed to parse op CLI JSON output: {e}") from e
236
237        except subprocess.TimeoutExpired:
238            self.logger.error(f"op CLI command timed out after {timeout_seconds}s")
239            raise BackendUnavailableError("op CLI command timed out") from None
240        except FileNotFoundError:
241            self.logger.error("op CLI not found in PATH")
242            raise BackendUnavailableError("op CLI not found") from None
243        finally:
244            # Update metrics
245            if self.enable_metrics:
246                success = result is not None and result.returncode == 0
247                self._update_metrics(latency_ms, success)

249    def _redact_sensitive_args(self, cmd: list[str]) -> list[str]:
250        """Redact sensitive information from command arguments for logging.
251
252        Args:
253            cmd: Command arguments to redact
254
255        Returns:
256            Command with sensitive information redacted
257        """
258        redacted = []
259        skip_next = False
260
261        for i, arg in enumerate(cmd):
262            if skip_next:
263                redacted.append("[REDACTED]")
264                skip_next = False
265            elif arg in ["--session", "-s"] and i + 1 < len(cmd):
266                redacted.append(arg)
267                skip_next = True
268            elif arg.startswith("--session=") or arg.startswith("-s="):
269                redacted.append(arg.split("=")[0] + "=[REDACTED]")
270            else:
271                redacted.append(arg)
272
273        return redacted

275    def _map_op_error(self, return_code: int, error_msg: str) -> SecretsManagerError | None:
276        """Map op CLI error codes and messages to appropriate SecretsManagerError subclasses.
277
278        Args:
279            return_code: The exit code from the op command
280            error_msg: The error message from stderr
281
282        Returns:
283            Appropriate SecretsManagerError subclass or None for unmapped errors
284        """
285        error_msg_lower = error_msg.lower()
286
287        # Common 1Password CLI error patterns
288        if return_code == 1:
289            if "not found" in error_msg_lower or "no such" in error_msg_lower:
290                return SecretNotFoundError(f"Secret not found: {error_msg}")
291            elif "permission denied" in error_msg_lower or "unauthorized" in error_msg_lower:
292                return PermissionDeniedError(f"Permission denied: {error_msg}")
293            elif "rate limit" in error_msg_lower or "too many requests" in error_msg_lower:
294                return RateLimitedError(f"Rate limited: {error_msg}")
295            elif "session expired" in error_msg_lower or "not signed in" in error_msg_lower:
296                return PermissionDeniedError(f"Session expired: {error_msg}")
297            elif "vault not found" in error_msg_lower:
298                return SecretNotFoundError(f"Vault not found: {error_msg}")
299            elif "item not found" in error_msg_lower:
300                return SecretNotFoundError(f"Item not found: {error_msg}")
301            elif "field not found" in error_msg_lower:
302                return SecretNotFoundError(f"Field not found: {error_msg}")
303
304        elif return_code == 2:
305            # Usually indicates authentication issues
306            return PermissionDeniedError(f"Authentication failed: {error_msg}")
307
308        elif return_code == 3:
309            # Usually indicates network or service issues
310            return BackendUnavailableError(f"Service unavailable: {error_msg}")
311
312        return None

378    def _run_op_command_with_retry(self, args: list[str], timeout: int | None = None) -> Any:
379        """Run op CLI command with retry logic for transient errors.
380
381        Args:
382            args: List of arguments to pass to op CLI
383            timeout: Override default timeout in seconds
384
385        Returns:
386            Parsed JSON output from the command
387
388        Raises:
389            SecretsManagerError: For various error conditions
390        """
391        last_exception = None
392
393        for attempt in range(self.max_retries + 1):
394            try:
395                if attempt > 0:
396                    # Check if we should retry based on the last exception
397                    if not self._should_retry(last_exception):  # type: ignore[arg-type]
398                        raise last_exception
399
400                    # Calculate exponential backoff delay
401                    delay = self.retry_delay_seconds * (2 ** (attempt - 1))
402                    self.logger.debug(
403                        f"Retrying op command in {delay}s (attempt {attempt + 1}/"
404                        f"{self.max_retries + 1})"
405                    )
406                    time.sleep(delay)
407
408                    # Update retry metrics
409                    if self.enable_metrics:
410                        self._metrics["retry_attempts"] += 1
411
412                return self._run_op_command_sync(args, timeout)
413
414            except (RateLimitedError, BackendUnavailableError) as e:
415                last_exception = e
416                if attempt == self.max_retries:
417                    self.logger.error(f"Max retries ({self.max_retries}) exceeded for op command")
418                    raise
419                else:
420                    self.logger.warning(f"Transient error on attempt {attempt + 1}: {e}")
421
422            except (SecretNotFoundError, PermissionDeniedError, InvalidReferenceError):
423                # These errors are not transient, don't retry
424                raise
425
426            except SecretsManagerError as e:
427                last_exception = e
428                if attempt == self.max_retries:
429                    raise
430                else:
431                    self.logger.warning(f"Error on attempt {attempt + 1}: {e}")
432
433        # This should never be reached, but just in case
434        raise last_exception or SecretsManagerError("Unknown error in retry logic")
435
436    def _should_retry(self, exception: Exception) -> bool:
437        """Determine if an operation should be retried based on the exception.
438
439        Args:
440            exception: The exception that occurred
441
442        Returns:
443            True if the operation should be retried, False otherwise
444        """
445        if isinstance(exception, RateLimitedError | BackendUnavailableError):
446            return True
447        return False

214
