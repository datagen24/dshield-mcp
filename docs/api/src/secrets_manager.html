239        """Check if a cache entry is still valid based on TTL.

242            cache_entry: Tuple of (value, timestamp)

244        Returns:

253    def _get_from_cache(self, key: str) -> Any | None:
254        """Retrieve a value from the cache if valid.
255

259        Returns:
260            Cached value if valid, None otherwise
261        """

133    """Represents a reference to a secret in a specific backend.
134    
135    This class handles the resolution and validation of secret reference URIs
136    like 'op://vault/item/field' for 1Password or 'vault://path/to/secret' for Vault.
137    
138    Attributes:
139        uri: The full reference URI
140        backend: The backend type (e.g., 'op', 'vault', 'aws')
141        vault: The vault or namespace identifier
142        item: The item or secret identifier
143        field: The specific field within the item (optional)
144        is_valid: Whether the reference URI is valid
145    """
146    uri: str
147    backend: str | None = None
148    vault: str | None = None
149    item: str | None = None
150    field: str | None = None
151    is_valid: bool = False
152    
153    def __post_init__(self) -> None:
154        """Parse and validate the reference URI after initialization."""
155        self._parse_uri()
156    
157    def _parse_uri(self) -> None:
158        """Parse the reference URI and extract components.
159        
160        Supports formats like:
161        - op://vault/item/field
162        - vault://path/to/secret
163        - aws://region/secret-name
164        """
165        if not self.uri:
166            self.is_valid = False
167            return
168            
169        # Pattern for op://vault/item/field format
170        op_pattern = r'^op://([^/]+)/([^/]+)(?:/(.+))?$'
171        op_match = re.match(op_pattern, self.uri)
172        if op_match:
173            self.backend = 'op'
174            self.vault = op_match.group(1)
175            self.item = op_match.group(2)
176            self.field = op_match.group(3)
177            self.is_valid = True
178            return
179            
180        # Pattern for vault://path/to/secret format
181        vault_pattern = r'^vault://([^/]+(?:/.*)?)$'
182        vault_match = re.match(vault_pattern, self.uri)
183        if vault_match:
184            self.backend = 'vault'
185            self.item = vault_match.group(1)
186            self.is_valid = True
187            return
188            
189        # Pattern for aws://region/secret-name format
190        aws_pattern = r'^aws://([^/]+)/(.+)$'
191        aws_match = re.match(aws_pattern, self.uri)

369    def _run_op_command_with_retry(self, args: list[str], timeout: int | None = None) -> Any:
370        """Run op CLI command with retry logic for transient errors.
371        
372        Args:
373            args: List of arguments to pass to op CLI
374            timeout: Override default timeout in seconds
375            
376        Returns:
377            Parsed JSON output from the command
378            
379        Raises:
380            SecretsManagerError: For various error conditions
381        """
382        last_exception = None
383        
384        for attempt in range(self.max_retries + 1):
385            try:
386                if attempt > 0:
387                    # Check if we should retry based on the last exception
388                    if not self._should_retry(last_exception):
389                        raise last_exception
390                    
391                    # Calculate exponential backoff delay
392                    delay = self.retry_delay_seconds * (2 ** (attempt - 1))
393                    self.logger.debug(f"Retrying op command in {delay}s (attempt {attempt + 1}/{self.max_retries + 1})")
394                    time.sleep(delay)
395                    
396                    # Update retry metrics
397                    if self.enable_metrics:
398                        self._metrics["retry_attempts"] += 1
399                
400                return self._run_op_command_sync(args, timeout)
401                
402            except (RateLimitedError, BackendUnavailableError) as e:
403                last_exception = e
404                if attempt == self.max_retries:
405                    self.logger.error(f"Max retries ({self.max_retries}) exceeded for op command")
406                    raise
407                else:
408                    self.logger.warning(f"Transient error on attempt {attempt + 1}: {e}")
409                    
410            except (SecretNotFoundError, PermissionDeniedError, InvalidReferenceError):
411                # These errors are not transient, don't retry
412                raise
413                
414            except SecretsManagerError as e:
415                last_exception = e
416                if attempt == self.max_retries:
417                    raise
418                else:
419                    self.logger.warning(f"Error on attempt {attempt + 1}: {e}")
420        
421        # This should never be reached, but just in case
422        raise last_exception or SecretsManagerError("Unknown error in retry logic")
