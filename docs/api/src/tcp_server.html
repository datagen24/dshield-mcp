 51    ) -> dict[str, Any] | None:
 52        """Process an MCP protocol message.
 53
 54        Args:
 55            connection: The TCP connection
 56            message: The MCP message to process
 57
 58        Returns:
 59            Response message if applicable, None for notifications
 60
 61        """
 62        try:
 63            # Check if this is an authentication message
 64            if message.get("method") == "authenticate":
 65                return await self._handle_authentication(connection, message)
 66
 67            # Check if connection is authenticated
 68            if not connection.is_authenticated:
 69                return self._create_error_response(
 70                    message.get("id"),
 71                    -32001,  # RESOURCE_NOT_FOUND
 72                    "Authentication required",
 73                    {"required": "authenticate"},
 74                )
 75
 76            # Route message to appropriate MCP handler
 77            return await self._route_mcp_message(connection, message)
 78
 79        except Exception as e:
 80            self.logger.error(
 81                "Error processing MCP message",
 82                client_address=connection.client_address,
 83                error=str(e),
 84            )
 85            return self._create_error_response(
 86                message.get("id"),
 87                -32603,  # INTERNAL_ERROR
 88                "Internal server error",
 89                {"error": str(e)},

149    async def _route_mcp_message(
150        self, connection: TCPConnection, message: dict[str, Any]
151    ) -> dict[str, Any] | None:
152        """Route MCP message to appropriate handler.
153
154        Args:
155            connection: The TCP connection
156            message: The MCP message
157
158        Returns:
159            Response message if applicable
160
161        """
162        method = message.get("method")
163        message_id = message.get("id")
164        params = message.get("params", {})
165
166        try:
167            # Handle different MCP methods
168            if method == "initialize":
169                return await self._handle_initialize(connection, message_id, params)
170            if method == "initialized":
171                return await self._handle_initialized(connection, message_id, params)
172            if method == "tools/list":
173                return await self._handle_tools_list(connection, message_id, params)
174            if method == "tools/call":
175                return await self._handle_tools_call(connection, message_id, params)
176            if method == "resources/list":
177                return await self._handle_resources_list(connection, message_id, params)
178            if method == "resources/read":
179                return await self._handle_resources_read(connection, message_id, params)
180            if method == "prompts/list":
181                return await self._handle_prompts_list(connection, message_id, params)
182            if method == "prompts/get":
183                return await self._handle_prompts_get(connection, message_id, params)
184            return self._create_error_response(
185                message_id,
186                -32601,  # METHOD_NOT_FOUND
187                f"Method '{method}' not found",
188                {"method": method},
189            )
190
191        except Exception as e:
192            self.logger.error(
193                "Error routing MCP message",
194                method=method,
195                client_address=connection.client_address,
196                error=str(e),
197            )
198            return self._create_error_response(
199                message_id,
200                -32603,  # INTERNAL_ERROR
201                "Error processing request",
202                {"error": str(e)},
203            )

502    This class provides a complete TCP server implementation that integrates

508        """Initialize the enhanced TCP server.
509
510        Args:
511            mcp_server: The MCP server instance
512            config: Server configuration
513
514        """
515        self.mcp_server = mcp_server
516        self.config = config or {}
517        self.logger = structlog.get_logger(f"{__name__}.{self.__class__.__name__}")
518
519        # Initialize components
520        self.connection_manager = ConnectionManager(self.config.get("connection_management", {}))
521        self.security_manager = TCPSecurityManager(self.config.get("security", {}))
522        self.authenticator = TCPAuthenticator(
523            self.connection_manager,
524            MCPErrorHandler(ErrorHandlingConfig()),
525            self.config.get("authentication", {}),
526        )
527        self.mcp_adapter = MCPServerAdapter(mcp_server, self.config.get("mcp_adapter", {}))
528
529        # Server state
530        self.is_running = False
531        self.server_socket: asyncio.Server | None = None
532        self.connections: set[TCPConnection] = set()
533        self._cleanup_task: asyncio.Task[None] | None = None

535    async def start(self) -> None:
536        """Start the enhanced TCP server.
537
538        Raises:
539            Exception: If server fails to start
540
541        """
542        try:
543            print(f"DEBUG: EnhancedTCPServer.start() called on port {self.config.get('port')}")
544            port = self.config.get("port", 3000)
545            bind_address = self.config.get("bind_address", "127.0.0.1")
546            max_connections = self.config.get("max_connections", 10)
547
548            self.logger.info(
549                "Starting enhanced TCP server",
550                port=port,
551                bind_address=bind_address,
552                max_connections=max_connections,
553            )
554
555            # Create TCP server
556            self.server_socket = await asyncio.start_server(
557                self._handle_connection,
558                bind_address,
559                port,
560                limit=max_connections,
561            )
562
563            # Start cleanup task
564            self._cleanup_task = asyncio.create_task(self._cleanup_expired_connections())
565
566            self.is_running = True
567            self.logger.info(
568                "Enhanced TCP server started successfully", port=port, bind_address=bind_address
569            )
570
571        except Exception as e:
572            print(f"ERROR starting server: {e}")
573            import traceback
574
575            self.logger.error(
576                "Failed to start enhanced TCP server",
577                error=str(e),
578                traceback=traceback.format_exc(),
579            )
580            raise  # Re-raise to see the error

582    async def stop(self) -> None:
583        """Stop the enhanced TCP server.
584
585        Closes all connections and cleans up resources.
586        """
587        try:
588            self.logger.info("Stopping enhanced TCP server")
589
590            # Cancel cleanup task
591            if self._cleanup_task:
592                self._cleanup_task.cancel()
593                try:
594                    await self._cleanup_task
595                except asyncio.CancelledError:
596                    pass
597
598            # Close all connections
599            for connection in list(self.connections):
600                await connection.close()
601            self.connections.clear()
602
603            # Close server socket
604            if self.server_socket:
605                self.server_socket.close()
606                await self.server_socket.wait_closed()
607
608            self.is_running = False
609            self.logger.info("Enhanced TCP server stopped successfully")

611        except Exception as e:
612            self.logger.error("Error stopping enhanced TCP server", error=str(e))

614    async def _handle_connection(
615        self, reader: asyncio.StreamReader, writer: asyncio.StreamWriter
616    ) -> None:
617        """Handle a new TCP connection.
618
619        Args:
620            reader: StreamReader for reading from the connection
621            writer: StreamWriter for writing to the connection
622
623        """
624        client_address = writer.get_extra_info("peername")
625        connection = None
626
627        try:
628            self.logger.info("New TCP connection", client_address=client_address)
629
630            # Check connection attempt limits
631            if not self.security_manager.record_connection_attempt(str(client_address)):
632                self.logger.warning("Connection attempt blocked", client_address=client_address)
633                writer.close()
634                await writer.wait_closed()
635                return
636
637            # Create connection object
638            connection = TCPConnection(reader, writer, client_address)
639            self.connections.add(connection)
640            self.connection_manager.add_connection(connection)
641
642            # Handle the connection
643            await self._process_connection(connection)
644
645        except Exception as e:
646            self.logger.error(
647                "Error handling TCP connection", client_address=client_address, error=str(e)
648            )
649        finally:
650            # Clean up connection
651            if connection:
652                self.connections.discard(connection)
653                self.connection_manager.remove_connection(connection)
654                await connection.close()
655            self.logger.info("TCP connection closed", client_address=client_address)

657    async def _process_connection(self, connection: TCPConnection) -> None:
658        """Process messages from a TCP connection.
659
660        Args:
661            connection: The TCP connection to process
662
663        """
664        try:
665            while True:
666                # Read message length (first 4 bytes)
667                length_data = await connection.reader.readexactly(4)
668                if not length_data:
669                    break
670
671                message_length = int.from_bytes(length_data, byteorder="big")
672
673                # Read message data
674                message_data = await connection.reader.readexactly(message_length)
675                if not message_data:
676                    break
677
678                # Parse JSON message
679                try:
680                    message = json.loads(message_data.decode("utf-8"))
681                    connection.update_activity()
682
683                    # Validate message with security manager
684                    validated_message = self.security_manager.validate_message(
685                        message,
686                        str(connection.client_address),
687                    )
688
689                    # Process message through MCP adapter
690                    response = await self.mcp_adapter.process_mcp_message(
691                        connection,
692                        validated_message,
693                    )
694
695                    # Send response if applicable
696                    if response:
697                        await self._send_response(connection, response)
698
699                except SecurityViolation as e:
700                    await self._send_error_response(connection, e.violation_type, str(e))
701                except json.JSONDecodeError as e:
702                    await self._send_error_response(
703                        connection, "INVALID_JSON", f"Invalid JSON: {e}"
704                    )
705                except Exception as e:
706                    await self._send_error_response(
707                        connection, "INTERNAL_ERROR", f"Internal error: {e}"
708                    )
709
710        except asyncio.IncompleteReadError:
711            # Connection closed by client
712            pass
713        except Exception as e:
714            self.logger.error(
715                "Error processing TCP connection",
716                client_address=connection.client_address,
717                error=str(e),
718            )

720    async def _send_response(self, connection: TCPConnection, response: dict[str, Any]) -> None:
721        """Send a response to a TCP connection.
722
723        Args:
724            connection: The TCP connection
725            response: The response to send
726
727        """
728        try:
729            message_data = json.dumps(response).encode("utf-8")
730            message_length = len(message_data)
731
732            # Send message length
733            connection.writer.write(message_length.to_bytes(4, byteorder="big"))
734            # Send message data
735            connection.writer.write(message_data)
736            await connection.writer.drain()
737
738        except Exception as e:
739            self.logger.error(
740                "Error sending TCP response", client_address=connection.client_address, error=str(e)
741            )

765    async def _cleanup_expired_connections(self) -> None:
766        """Clean up expired connections and security data.
767
768        Runs periodically to remove expired connections and clean up security data.
769        """
770        timeout_seconds = self.config.get("connection_timeout_seconds", 300)
771
772        while self.is_running:
773            try:
774                await asyncio.sleep(60)  # Check every minute
775
776                # Clean up expired connections
777                expired_connections = [
778                    conn for conn in self.connections if conn.is_expired(timeout_seconds)
779                ]
780
781                for connection in expired_connections:
782                    self.logger.info(
783                        "Closing expired connection", client_address=connection.client_address
784                    )
785                    self.connections.discard(connection)
786                    self.connection_manager.remove_connection(connection)
787                    await connection.close()
788
789                # Clean up security data
790                self.security_manager.cleanup_expired_data()
791
792                # Clean up expired API keys
793                self.connection_manager.cleanup_expired_keys()
794
795            except asyncio.CancelledError:
796                break
797            except Exception as e:
