148    def __init__(self, server: Any, config: dict[str, Any] | None = None) -> None:

152            server: The MCP server instance

154
155        """
156        super().__init__(server, config)

172        """Start the TCP transport.

174        Creates and binds the TCP server socket, then begins accepting connections.

176        Raises:
177            TransportError: If the TCP transport fails to start

212    async def stop(self) -> None:
213        """Stop the TCP transport.
214
215        Closes all connections, stops the server socket, and cleans up resources.
216        """
217        try:
218            self.logger.info("Stopping TCP transport")
219
220            # Cancel cleanup task
221            if self._cleanup_task:
222                self._cleanup_task.cancel()
223                try:
224                    await self._cleanup_task
225                except asyncio.CancelledError:
226                    pass
227
228            # Close all connections
229            for connection in list(self.connections):
230                await connection.close()
231            self.connections.clear()
232
233            # Close server socket
234            if self.server_socket:
235                self.server_socket.close()
236                await self.server_socket.wait_closed()
237
238            self.is_running = False
239            self.logger.info("TCP transport stopped successfully")
240
241        except Exception as e:
242            self.logger.error("Error stopping TCP transport", error=str(e))

245        """Run the TCP transport main loop.
246
247        Starts the server and waits for it to complete.
248
249        Raises:
250            TransportError: If the transport fails during execution
251
252        """

271        self, reader: asyncio.StreamReader, writer: asyncio.StreamWriter
272    ) -> None:
273        """Handle a new TCP connection.
274
275        Args:
276            reader: StreamReader for reading from the connection

278
279        """
280        client_address = writer.get_extra_info("peername")

282
283        try:
284            self.logger.info("New TCP connection", client_address=client_address)

286            # Create connection object
287            connection = TCPConnection(reader, writer, client_address)

293        except Exception as e:
294            self.logger.error(
295                "Error handling TCP connection", client_address=client_address, error=str(e)
296            )
297        finally:
298            # Clean up connection
299            if connection:
300                self.connections.discard(connection)
301                await connection.close()

304    async def _process_connection(self, connection: TCPConnection) -> None:
305        """Process messages from a TCP connection.
306
307        Args:
308            connection: The TCP connection to process
309
310        """
311        try:
312            while True:
313                # Read message length (first 4 bytes)
314                length_data = await connection.reader.readexactly(4)
315                if not length_data:
316                    break
317
318                message_length = int.from_bytes(length_data, byteorder="big")
319
320                # Read message data
321                message_data = await connection.reader.readexactly(message_length)
322                if not message_data:
323                    break
324
325                # Parse JSON message
326                try:
327                    message = json.loads(message_data.decode("utf-8"))
328                    connection.update_activity()
329
330                    # Check rate limiting
331                    if not connection.rate_limiter.is_allowed():
332                        await self._send_error_response(connection, -32008, "Rate limit exceeded")
333                        continue
334
335                    # Process message (placeholder - will be implemented with MCP protocol)
336                    await self._process_mcp_message(connection, message)
337
338                except json.JSONDecodeError as e:
339                    await self._send_error_response(connection, -32700, f"Invalid JSON: {e}")
340                except Exception as e:
341                    await self._send_error_response(connection, -32603, f"Internal error: {e}")
342
343        except asyncio.IncompleteReadError:
344            # Connection closed by client
345            pass
346        except Exception as e:
347            self.logger.error(
348                "Error processing TCP connection",
349                client_address=connection.client_address,
350                error=str(e),
