User Configuration Management for DShield MCP.

This module provides comprehensive user configuration management for the DShield MCP server. It extends the base configuration system with user-customizable settings, validation, environment variable support, and multiple configuration sources.

Features:

  • User-configurable settings with validation
  • Environment variable overrides
  • Multiple configuration file sources
  • Setting categories (query, pagination, streaming, etc.)
  • Configuration export and import
  • 1Password CLI integration

Example:

from src.user_config import get_user_config config = get_user_config() page_size = config.get_setting("query", "default_page_size") print(page_size)

 343class UserConfigManager:
 344    """Manages user-configurable settings with validation and environment variable support.
 345
 346    This class provides a comprehensive configuration management system that
 347    supports multiple configuration sources with precedence ordering:
 348    1. Environment variables (highest priority)
 349    2. User configuration file
 350    3. Base configuration
 351    4. Default values (lowest priority)
 352
 353    Attributes:
 354        config_path: Path to the configuration file
 355        op_secrets: OnePassword secrets manager
 356        base_config: Base configuration dictionary
 357        query_settings: Query-related settings
 358        pagination_settings: Pagination-related settings
 359        streaming_settings: Streaming-related settings
 360        performance_settings: Performance-related settings
 361        security_settings: Security-related settings
 362        logging_settings: Logging-related settings
 363        campaign_settings: Campaign analysis settings
 364        tcp_transport_settings: TCP transport settings
 365        api_key_management_settings: API key management settings
 366        tui_settings: TUI settings
 367        output_directory: Directory for generated outputs
 368            (default: ~/dshield-mcp-output, configurable)
 369
 370    Example:
 371        >>> manager = UserConfigManager()
 372        >>> output_dir = manager.output_directory
 373        >>> print(output_dir)
 374
 375    """
 376
 377    def __init__(self, config_path: str | None = None) -> None:
 378        """Initialize the UserConfigManager.
 379
 380        Args:
 381            config_path: Optional path to the configuration file
 382
 383        """
 384        self.config_path = config_path
 385        self.op_secrets = OnePasswordSecrets()
 386
 387        # Load base configuration
 388        try:
 389            self.base_config = get_config(config_path)
 390        except ConfigError as e:
 391            logger.warning(f"Failed to load base config: {e}")
 392            self.base_config = {}
 393
 394        # Initialize settings with defaults
 395        self.query_settings = QuerySettings()
 396        self.pagination_settings = PaginationSettings()
 397        self.streaming_settings = StreamingSettings()
 398        self.performance_settings = PerformanceSettings()
 399        self.security_settings = SecuritySettings()
 400        self.logging_settings = LoggingSettings()
 401        self.campaign_settings = CampaignSettings()
 402        self.tcp_transport_settings = TCPTransportSettings()
 403        self.api_key_management_settings = APIKeyManagementSettings()
 404        self.tui_settings = TUISettings()
 405
 406        # Output directory (default, can be overridden)
 407        self.output_directory = None  # type: Optional[str]
 408
 409        # Load user configuration
 410        self._load_user_config()
 411
 412        # Ensure output directory exists
 413        if self.output_directory is None:
 414            self.output_directory = os.path.expanduser(
 415                os.getenv("DMC_OUTPUT_DIRECTORY", "~/dshield-mcp-output")
 416            )
 417        self.output_directory = os.path.expandvars(self.output_directory)
 418        self.output_directory = os.path.abspath(self.output_directory)
 419        os.makedirs(self.output_directory, exist_ok=True)

 421    def _load_user_config(self) -> None:
 422        """Load user configuration from multiple sources with precedence.
 423
 424        Loads configuration from environment variables, user config file,
 425        and base config, applying them in order of precedence.
 426        """
 427        # Priority order: environment variables > user config file > base config > defaults
 428
 429        # Load from user config file
 430        user_config = self._load_user_config_file()
 431
 432        # Output directory (YAML root key)
 433        if "output_directory" in user_config:
 434            self.output_directory = user_config["output_directory"]
 435
 436        # Apply environment variable overrides
 437        self._apply_env_overrides()
 438
 439        # Apply user config file settings
 440        self._apply_user_config(user_config)
 441
 442        # Validate all settings
 443        self._validate_settings()
 444
 445    def _load_user_config_file(self) -> dict[str, Any]:
 446        """Load user configuration from file.
 447
 448        If a specific config path was provided, use that. Otherwise, searches for
 449        user configuration files in multiple locations:
 450        - Current directory: user_config.yaml
 451        - Config directory: config/user_config.yaml
 452        - Home directory: ~/.dshield-mcp/user_config.yaml
 453
 454        Returns:
 455            Dictionary containing user configuration or empty dict if not found
 456
 457        """
 458        # If a specific config path was provided, use that first
 459        if self.config_path:
 460            config_path = Path(self.config_path)
 461            if config_path.exists():
 462                try:
 463                    with open(config_path) as f:
 464                        config = yaml.safe_load(f)
 465                    logger.info(f"Loaded user config from: {config_path}")
 466                    return config or {}
 467                except Exception as e:
 468                    logger.warning(f"Failed to load user config from {config_path}: {e}")
 469            else:
 470                logger.warning(f"Specified config file does not exist: {config_path}")
 471
 472        # Fall back to default search paths
 473        user_config_paths = [
 474            Path("user_config.yaml"),
 475            Path("config/user_config.yaml"),
 476            Path.home() / ".dshield-mcp" / "user_config.yaml",
 477        ]
 478
 479        for config_path in user_config_paths:
 480            if config_path.exists():
 481                try:
 482                    with open(config_path) as f:
 483                        config = yaml.safe_load(f)
 484                    logger.info(f"Loaded user config from: {config_path}")
 485                    return config or {}
 486                except Exception as e:
 487                    logger.warning(f"Failed to load user config from {config_path}: {e}")
 488
 489        return {}

1027            errors.append("log_format must be one of: json, text")
