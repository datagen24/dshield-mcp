User Configuration Management for DShield MCP.

This module provides comprehensive user configuration management for the DShield MCP server. It extends the base configuration system with user-customizable settings, validation, environment variable support, and multiple configuration sources.

Features:

  • User-configurable settings with validation
  • Environment variable overrides
  • Multiple configuration file sources
  • Setting categories (query, pagination, streaming, etc.)
  • Configuration export and import
  • 1Password CLI integration

Example:

from src.user_config import get_user_config config = get_user_config() page_size = config.get_setting("query", "default_page_size") print(page_size)

 343class UserConfigManager:
 344    """Manages user-configurable settings with validation and environment variable support.
 345
 346    This class provides a comprehensive configuration management system that
 347    supports multiple configuration sources with precedence ordering:
 348    1. Environment variables (highest priority)
 349    2. User configuration file
 350    3. Base configuration
 351    4. Default values (lowest priority)
 352
 353    Attributes:
 354        config_path: Path to the configuration file
 355        op_secrets: OnePassword secrets manager
 356        base_config: Base configuration dictionary
 357        query_settings: Query-related settings
 358        pagination_settings: Pagination-related settings
 359        streaming_settings: Streaming-related settings
 360        performance_settings: Performance-related settings
 361        security_settings: Security-related settings
 362        logging_settings: Logging-related settings
 363        campaign_settings: Campaign analysis settings
 364        tcp_transport_settings: TCP transport settings
 365        api_key_management_settings: API key management settings
 366        tui_settings: TUI settings
 367        output_directory: Directory for generated outputs (default: ~/dshield-mcp-output, configurable)
 368
 369    Example:
 370        >>> manager = UserConfigManager()
 371        >>> output_dir = manager.output_directory
 372        >>> print(output_dir)
 373
 374    """
 375
 376    def __init__(self, config_path: str | None = None) -> None:
 377        """Initialize the UserConfigManager.
 378
 379        Args:
 380            config_path: Optional path to the configuration file
 381
 382        """
 383        self.config_path = config_path
 384        self.op_secrets = OnePasswordSecrets()
 385
 386        # Load base configuration
 387        try:
 388            self.base_config = get_config(config_path)
 389        except ConfigError as e:
 390            logger.warning(f"Failed to load base config: {e}")
 391            self.base_config = {}
 392
 393        # Initialize settings with defaults
 394        self.query_settings = QuerySettings()
 395        self.pagination_settings = PaginationSettings()
 396        self.streaming_settings = StreamingSettings()
 397        self.performance_settings = PerformanceSettings()
 398        self.security_settings = SecuritySettings()
 399        self.logging_settings = LoggingSettings()
 400        self.campaign_settings = CampaignSettings()
 401        self.tcp_transport_settings = TCPTransportSettings()
 402        self.api_key_management_settings = APIKeyManagementSettings()
 403        self.tui_settings = TUISettings()
 404
 405        # Output directory (default, can be overridden)
 406        self.output_directory = None  # type: Optional[str]
 407
 408        # Load user configuration
 409        self._load_user_config()
 410
 411        # Ensure output directory exists
 412        if self.output_directory is None:
 413            self.output_directory = os.path.expanduser(
 414                os.getenv("DMC_OUTPUT_DIRECTORY", "~/dshield-mcp-output")
 415            )
 416        self.output_directory = os.path.expandvars(self.output_directory)
 417        self.output_directory = os.path.abspath(self.output_directory)
 418        os.makedirs(self.output_directory, exist_ok=True)

 420    def _load_user_config(self) -> None:
 421        """Load user configuration from multiple sources with precedence.
 422
 423        Loads configuration from environment variables, user config file,
 424        and base config, applying them in order of precedence.
 425        """
 426        # Priority order: environment variables > user config file > base config > defaults
 427
 428        # Load from user config file
 429        user_config = self._load_user_config_file()
 430
 431        # Output directory (YAML root key)
 432        if "output_directory" in user_config:
 433            self.output_directory = user_config["output_directory"]
 434
 435        # Apply environment variable overrides
 436        self._apply_env_overrides()
 437
 438        # Apply user config file settings
 439        self._apply_user_config(user_config)
 440
 441        # Validate all settings
 442        self._validate_settings()

 444    def _load_user_config_file(self) -> dict[str, Any]:
 445        """Load user configuration from file.
 446
 447        If a specific config path was provided, use that. Otherwise, searches for
 448        user configuration files in multiple locations:
 449        - Current directory: user_config.yaml
 450        - Config directory: config/user_config.yaml
 451        - Home directory: ~/.dshield-mcp/user_config.yaml
 452
 453        Returns:
 454            Dictionary containing user configuration or empty dict if not found
 455
 456        """
 457        # If a specific config path was provided, use that first
 458        if self.config_path:
 459            config_path = Path(self.config_path)
 460            if config_path.exists():
 461                try:
 462                    with open(config_path) as f:
 463                        config = yaml.safe_load(f)
 464                    logger.info(f"Loaded user config from: {config_path}")
 465                    return config or {}
 466                except Exception as e:
 467                    logger.warning(f"Failed to load user config from {config_path}: {e}")
 468            else:
 469                logger.warning(f"Specified config file does not exist: {config_path}")
 470
 471        # Fall back to default search paths
 472        user_config_paths = [
 473            Path("user_config.yaml"),
 474            Path("config/user_config.yaml"),
 475            Path.home() / ".dshield-mcp" / "user_config.yaml",
 476        ]
 477
 478        for config_path in user_config_paths:
 479            if config_path.exists():
 480                try:
 481                    with open(config_path) as f:
 482                        config = yaml.safe_load(f)
 483                    logger.info(f"Loaded user config from: {config_path}")
 484                    return config or {}
 485                except Exception as e:
 486                    logger.warning(f"Failed to load user config from {config_path}: {e}")
 487
 488        return {}
